'	строка вида "0660BqCps" возвращает поверхностную активность для 6130cps

	'msgbox "get_activity_Sr - "+src
	select case src
	'плотность потока
		case "0660": get_activity_Sr = BETA_0660.getFDensity(nowDate)

		case "0667": get_activity_Sr = BETA_0667.getFDensity(nowDate)

		case "2125": get_activity_Sr = BETA_2125.getFDensity(nowDate)
		case "2126": get_activity_Sr = BETA_2126.getFDensity(nowDate)
		case "2127": get_activity_Sr = BETA_2127.getFDensity(nowDate)
		case "2128": get_activity_Sr = BETA_2128.getFDensity(nowDate)				
		case "2129": get_activity_Sr = BETA_2129.getFDensity(nowDate)
		case "0578": get_activity_Sr = BETA_0578.getFDensity(nowDate)	
		case "0579": get_activity_Sr = BETA_0579.getFDensity(nowDate)	
		case "0580": get_activity_Sr = BETA_0580.getFDensity(nowDate)	
		case "0581": get_activity_Sr = BETA_0581.getFDensity(nowDate)				
	'—————————————————————————————————————————————————————————————————
	'поверхностная активность
		case "0660Bq": get_activity_Sr = BETA_0660.getSurActivity(nowDate)
        case "0661Bq": get_activity_Sr = BETA_0661.getSurActivity(nowDate)
        case "0662Bq": get_activity_Sr = BETA_0662.getSurActivity(nowDate)
        case "0663Bq": get_activity_Sr = BETA_0663.getSurActivity(nowDate)
        case "0664Bq": get_activity_Sr = BETA_0664.getSurActivity(nowDate)
        case "0665Bq": get_activity_Sr = BETA_0665.getSurActivity(nowDate)
        case "0666Bq": get_activity_Sr = BETA_0666.getSurActivity(nowDate)
        case "0667Bq": get_activity_Sr = BETA_0667.getSurActivity(nowDate)
        case "2124Bq": get_activity_Sr = BETA_2124.getSurActivity(nowDate)
        case "2125Bq": get_activity_Sr = BETA_2125.getSurActivity(nowDate)
        case "2126Bq": get_activity_Sr = BETA_2126.getSurActivity(nowDate)
        case "2127Bq": get_activity_Sr = BETA_2127.getSurActivity(nowDate)
        case "2128Bq": get_activity_Sr = BETA_2128.getSurActivity(nowDate)				
        case "2129Bq": get_activity_Sr = BETA_2129.getSurActivity(nowDate)
        case "0578Bq": get_activity_Sr = BETA_0578.getSurActivity(nowDate)	
        case "0579Bq": get_activity_Sr = BETA_0579.getSurActivity(nowDate)	
        case "0580Bq": get_activity_Sr = BETA_0580.getSurActivity(nowDate)	
        case "0581Bq": get_activity_Sr = BETA_0581.getSurActivity(nowDate)
	'—————————————————————————————————————————————————————————————————
	'активность (нигде не используется, так, на всякий)
		case "0660Act": get_activity_Sr = BETA_0660.getActivity(nowDate)
        case "0661Act": get_activity_Sr = BETA_0661.getActivity(nowDate)
        case "0662Act": get_activity_Sr = BETA_0662.getActivity(nowDate)
        case "0663Act": get_activity_Sr = BETA_0663.getActivity(nowDate)
        case "0664Act": get_activity_Sr = BETA_0664.getActivity(nowDate)
        case "0665Act": get_activity_Sr = BETA_0665.getActivity(nowDate)
        case "0666Act": get_activity_Sr = BETA_0666.getActivity(nowDate)
        case "0667Act": get_activity_Sr = BETA_0667.getActivity(nowDate)
        case "2124Act": get_activity_Sr = BETA_2124.getActivity(nowDate)
        case "2125Act": get_activity_Sr = BETA_2125.getActivity(nowDate)

		case "0665/S": get_activity_Sr = BETA_0665.getFDensityB3(nowDate)

		case "0661BqCps": get_activity_Sr = BETA_0661.getCpsActivity(nowDate)
		case "0662BqCps": get_activity_Sr = BETA_0662.getCpsActivity(nowDate)
		case "0578BqCps": get_activity_Sr = BETA_0578.getCpsActivity(nowDate)
